for (i in 6:12) {
    # innerhalb des for()-loops die Nummern der
    # gewuenschten Spalten einstellen
    d <- DF_mod_day %>%
        pull(i)
    d <- density(d)
    dp <- DF_mod_day %>%
        filter(pres_abs == 1) %>%
        pull(i)
    dp <- density(dp)
    da <- DF_mod_day %>%
        filter(pres_abs == 0) %>%
        pull(i)
    da <- density(da)
    plot(0, 0, type = "l", xlim = range(c(dp$x, da$x)), ylim = range(dp$y,
        da$y), xlab = names(DF_mod_day[i]), ylab = "Density")
    lines(dp$x, dp$y, col = "blue")  # Praesenz
    lines(da$x, da$y, col = "red")  # Absenz
}

### Aufgabe 3

Erstellen von Density Plots der Präsenz / Absenz in Abhängigkeit der unabhängigen Variablen

```{r tidy=TRUE, results="hide", eval=FALSE, fig.keep="none"}
# Ein Satz Density Plots für den Tagesdatensatz und einer für den Nachtdatensatz 

par(mfrow=c(3,3), mar=c(4, 4, 3, 3))
for (i in 6:12) {          # innerhalb des for()-loops die Nummern der gewünschten 
                           # Spalten einstellen
  d  <-  DF_mod_day %>% pull(i)
  d  <-  density(d)
  dp  <-  DF_mod_day %>% filter(pres_abs == 1) %>% pull(i)
  dp <- density(dp)
  da  <-  DF_mod_day %>% filter(pres_abs == 0) %>% pull(i)
  da <- density(da)
  plot(0,0, type="l", xlim=range(c(dp$x,da$x)), ylim=range(dp$y,da$y), 
       xlab=names(DF_mod_day[i]), ylab="Density")
  lines(dp$x, dp$y, col="blue")             # Präsenz
  lines(da$x, da$y, col="red")              # Absenz
}
```
